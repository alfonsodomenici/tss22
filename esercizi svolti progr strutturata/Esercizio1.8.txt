start
	x = leggi numero;

	n = 1;
	sommaDivisori = 0;
	while(n < x) do
		if(x/n resto 0) then
			sommaDivisori = sommaDivisori + n;
		end if
		n = n+1;
	end while

	if(sommaDivisori == x) then
		stampa  perfetto;
	else if(sommaDivisori > x) then
		stampa abbondante;
	else
		stampa difettivo;
	end if 

end
